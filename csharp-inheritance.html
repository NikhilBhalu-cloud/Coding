Inheritance in C#

Inheritance, method overriding, base keyword, and inheritance hierarchies

Understanding Inheritance

Creating new classes based on existing ones

Inheritance is a fundamental OOP principle that allows a class to inherit properties and methods from another class. This promotes code reuse and establishes an "is-a" relationship between classes.

Key Concepts
  • Base Class (Parent): The class being inherited from
  • Derived Class (Child): The class that inherits
  • Method Overriding: Redefining base class methods
  • Virtual/Override: Keywords for polymorphism
Benefits
  • Code reuse and reduction of duplication
  • Establishing hierarchical relationships
  • Enabling polymorphism
  • Easier maintenance and updates

Basic Inheritance

Creating derived classes from a base class

using System;

// Base class (Parent)
public class Animal
{
    public string Name { get; protected set; }
    public int Age { get; protected set; }
    
    public Animal(string name, int age)
    {
        Name = name;
        Age = age;
    }
    
    // Virtual method - can be overridden
    public virtual void MakeSound()
    {
        Console.WriteLine($"{Name} makes a generic animal sound");
    }
    
    // Regular method - inherited as-is
    public void Eat()
    {
        Console.WriteLine($"{Name} is eating");
    }
    
    public virtual void DisplayInfo()
    {
        Console.WriteLine($"Animal: {Name}, Age: {Age}");
    }
}

// Derived class (Child)
public class Dog : Animal
{
    public string Breed { get; private set; }
    
    public Dog(string name, int age, string breed) : base(name, age)
    {
        Breed = breed;
    }
    
    // Override the virtual method
    public override void MakeSound()
    {
        Console.WriteLine($"{Name} barks: Woof! Woof!");
    }
    
    // Override with additional functionality
    public override void DisplayInfo()
    {
        base.DisplayInfo(); // Call base class method
        Console.WriteLine($"Breed: {Breed}");
    }
    
    // Dog-specific method
    public void Fetch()
    {
        Console.WriteLine($"{Name} is fetching the ball");
    }
}

// Another derived class
public class Cat : Animal
{
    public string Color { get; private set; }
    
    public Cat(string name, int age, string color) : base(name, age)
    {
        Color = color;
    }
    
    public override void MakeSound()
    {
        Console.WriteLine($"{Name} meows: Meow! Meow!");
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Color: {Color}");
    }
    
    public void Climb()
    {
        Console.WriteLine($"{Name} is climbing a tree");
    }
}

class InheritanceExample
{
    static void Main()
    {
        Console.WriteLine("=== Basic Inheritance ===");
        
        // Create instances
        Animal animal = new Animal("Generic Animal", 5);
        Dog dog = new Dog("Buddy", 3, "Golden Retriever");
        Cat cat = new Cat("Whiskers", 2, "Orange");
        
        // Polymorphic behavior
        Animal[] animals = { animal, dog, cat };
        
        foreach (Animal a in animals)
        {
            a.DisplayInfo();
            a.MakeSound(); // Calls appropriate overridden method
            a.Eat();       // Inherited method
            Console.WriteLine();
        }
        
        // Specific behavior
        dog.Fetch();
        cat.Climb();
    }
}

Method Overriding

Redefining base class methods in derived classes

using System;

// Abstract base class for shapes
public abstract class Shape
{
    public string Name { get; protected set; }
    
    public Shape(string name)
    {
        Name = name;
    }
    
    // Virtual method - can be overridden
    public virtual void DisplayInfo()
    {
        Console.WriteLine($"Shape: {Name}");
    }
    
    // Abstract methods - must be implemented
    public abstract double CalculateArea();
    public abstract double CalculatePerimeter();
}

public class Circle : Shape
{
    public double Radius { get; private set; }
    
    public Circle(double radius) : base("Circle")
    {
        Radius = radius;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Radius: {Radius}");
    }
    
    public override double CalculateArea()
    {
        return Math.PI * Radius * Radius;
    }
    
    public override double CalculatePerimeter()
    {
        return 2 * Math.PI * Radius;
    }
}

public class Rectangle : Shape
{
    public double Width { get; private set; }
    public double Height { get; private set; }
    
    public Rectangle(double width, double height) : base("Rectangle")
    {
        Width = width;
        Height = height;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Width: {Width}, Height: {Height}");
    }
    
    public override double CalculateArea()
    {
        return Width * Height;
    }
    
    public override double CalculatePerimeter()
    {
        return 2 * (Width + Height);
    }
}

class MethodOverridingExample
{
    static void Main()
    {
        Console.WriteLine("=== Method Overriding ===");
        
        Shape[] shapes = {
            new Circle(5),
            new Rectangle(4, 6)
        };
        
        foreach (Shape shape in shapes)
        {
            shape.DisplayInfo();      // Virtual method
            Console.WriteLine($"Area: {shape.CalculateArea():F2}");
            Console.WriteLine($"Perimeter: {shape.CalculatePerimeter():F2}");
            Console.WriteLine();
        }
    }
}

The base Keyword

Accessing base class members from derived classes

using System;

public class Employee
{
    public string Name { get; protected set; }
    public int Id { get; protected set; }
    public decimal Salary { get; protected set; }
    
    public Employee(string name, int id, decimal salary)
    {
        Name = name;
        Id = id;
        Salary = salary;
    }
    
    public virtual void DisplayInfo()
    {
        Console.WriteLine($"Employee: {Name} (ID: {Id})");
        Console.WriteLine($"Salary: ${Salary:N2}");
    }
    
    public virtual decimal CalculateBonus()
    {
        return Salary * 0.05m; // 5% of salary
    }
}

public class Manager : Employee
{
    public string Department { get; private set; }
    public int TeamSize { get; private set; }
    
    public Manager(string name, int id, decimal salary, string department, int teamSize)
        : base(name, id, salary) // Call base constructor
    {
        Department = department;
        TeamSize = teamSize;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo(); // Call base method
        Console.WriteLine($"Department: {Department}");
        Console.WriteLine($"Team Size: {TeamSize}");
        Console.WriteLine($"Bonus: ${CalculateBonus():N2}");
    }
    
    public override decimal CalculateBonus()
    {
        // Base bonus + team bonus
        decimal baseBonus = base.CalculateBonus();
        decimal teamBonus = TeamSize * 1000m;
        return baseBonus + teamBonus;
    }
    
    public void ConductMeeting()
    {
        Console.WriteLine($"{Name} is conducting a meeting for {Department} team");
    }
}

public class Developer : Employee
{
    public string ProgrammingLanguage { get; private set; }
    public int ProjectsCompleted { get; private set; }
    
    public Developer(string name, int id, decimal salary, string language, int projects)
        : base(name, id, salary)
    {
        ProgrammingLanguage = language;
        ProjectsCompleted = projects;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Programming Language: {ProgrammingLanguage}");
        Console.WriteLine($"Projects Completed: {ProjectsCompleted}");
    }
    
    public override decimal CalculateBonus()
    {
        // Base bonus + project bonus
        return base.CalculateBonus() + (ProjectsCompleted * 500m);
    }
    
    public void WriteCode()
    {
        Console.WriteLine($"{Name} is writing {ProgrammingLanguage} code");
    }
}

class BaseKeywordExample
{
    static void Main()
    {
        Console.WriteLine("=== Base Keyword Example ===");
        
        Employee[] employees = {
            new Employee("John Doe", 101, 50000),
            new Manager("Jane Smith", 102, 80000, "Engineering", 8),
            new Developer("Bob Johnson", 103, 70000, "C#", 12)
        };
        
        foreach (Employee emp in employees)
        {
            emp.DisplayInfo();
            Console.WriteLine();
        }
    }
}

Multilevel Inheritance

Creating inheritance chains with multiple levels

using System;

// Base class (Level 1)
public class Vehicle
{
    public string Make { get; protected set; }
    public string Model { get; protected set; }
    public int Year { get; protected set; }
    
    public Vehicle(string make, string model, int year)
    {
        Make = make;
        Model = model;
        Year = year;
    }
    
    public virtual void DisplayInfo()
    {
        Console.WriteLine($"Vehicle: {Year} {Make} {Model}");
    }
    
    public virtual void Start()
    {
        Console.WriteLine("Vehicle is starting...");
    }
}

// Derived class (Level 2)
public class Car : Vehicle
{
    public int NumberOfDoors { get; protected set; }
    
    public Car(string make, string model, int year, int doors)
        : base(make, model, year)
    {
        NumberOfDoors = doors;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Type: Car, Doors: {NumberOfDoors}");
    }
    
    public override void Start()
    {
        Console.WriteLine("Car engine is starting...");
    }
    
    public virtual void Drive()
    {
        Console.WriteLine("Car is driving on the road");
    }
}

// Derived class (Level 3)
public class SportsCar : Car
{
    public int Horsepower { get; private set; }
    public bool HasTurbo { get; private set; }
    
    public SportsCar(string make, string model, int year, int doors, int horsepower, bool turbo)
        : base(make, model, year, doors)
    {
        Horsepower = horsepower;
        HasTurbo = turbo;
    }
    
    public override void DisplayInfo()
    {
        base.DisplayInfo();
        Console.WriteLine($"Type: Sports Car, Horsepower: {Horsepower}, Turbo: {HasTurbo}");
    }
    
    public override void Drive()
    {
        Console.WriteLine("Sports car is racing at high speed!");
    }
    
    public void ActivateTurbo()
    {
        if (HasTurbo)
        {
            Console.WriteLine("Turbo activated! Extra power engaged!");
        }
        else
        {
            Console.WriteLine("This car doesn't have turbo");
        }
    }
}

class MultilevelInheritanceExample
{
    static void Main()
    {
        Console.WriteLine("=== Multilevel Inheritance ===");
        
        // Create instances at different levels
        Vehicle vehicle = new Vehicle("Generic", "Vehicle", 2020);
        Car car = new Car("Toyota", "Camry", 2022, 4);
        SportsCar sportsCar = new SportsCar("Ferrari", "F8", 2023, 2, 710, true);
        
        // Demonstrate inheritance chain
        Vehicle[] vehicles = { vehicle, car, sportsCar };
        
        foreach (Vehicle v in vehicles)
        {
            v.DisplayInfo();
            v.Start();
            
            // Type-specific operations
            if (v is Car carObj)
            {
                carObj.Drive();
            }
            
            if (v is SportsCar sportsCarObj)
            {
                sportsCarObj.ActivateTurbo();
            }
            
            Console.WriteLine();
        }
    }
}

Sealed Classes and Methods

Preventing further inheritance

using System;

public class BaseCalculator
{
    public virtual double Add(double a, double b)
    {
        return a + b;
    }
    
    public virtual double Multiply(double a, double b)
    {
        return a * b;
    }
}

public class AdvancedCalculator : BaseCalculator
{
    // Sealed method - cannot be overridden further
    public sealed override double Add(double a, double b)
    {
        Console.WriteLine("Advanced addition with logging");
        return base.Add(a, b);
    }
    
    public override double Multiply(double a, double b)
    {
        Console.WriteLine("Advanced multiplication");
        return base.Multiply(a, b);
    }
    
    public virtual double Power(double baseNum, double exponent)
    {
        return Math.Pow(baseNum, exponent);
    }
}

// Sealed class - cannot be inherited
public sealed class FinalCalculator : AdvancedCalculator
{
    // This would cause compilation error because Add is sealed:
    // public override double Add(double a, double b) { ... }
    
    public override double Multiply(double a, double b)
    {
        Console.WriteLine("Final multiplication implementation");
        return base.Multiply(a, b);
    }
    
    public override double Power(double baseNum, double exponent)
    {
        Console.WriteLine("Optimized power calculation");
        return base.Power(baseNum, exponent);
    }
}

// This would cause compilation error because FinalCalculator is sealed:
// public class UltimateCalculator : FinalCalculator { ... }

class SealedExample
{
    static void Main()
    {
        Console.WriteLine("=== Sealed Classes and Methods ===");
        
        FinalCalculator calc = new FinalCalculator();
        
        Console.WriteLine($"Add: {calc.Add(5, 3)}");
        Console.WriteLine($"Multiply: {calc.Multiply(4, 6)}");
        Console.WriteLine($"Power: {calc.Power(2, 8)}");
    }
}
