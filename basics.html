Variables & Data Types

Value Types vs Reference Types

C# has two main categories of types:

  • Value Types: Store data directly (int, double, bool, char, struct, enum)
  • Reference Types: Store references to data (string, object, class, interface, delegate) 
// Value Types
int age = 25;
double salary = 50000.50;
bool isActive = true;
char grade = 'A';

// Reference Types
string name = "John Doe";
object obj = new object();
int[] numbers = {1, 2, 3, 4, 5};

// Type inference with var
var message = "Hello World"; // Compiler infers string
var count = 10; // Compiler infers int
Constants and Read-only Fields
// Constants - compile-time constants
const double PI = 3.14159;
const string COMPANY_NAME = "TechCorp";

// Read-only fields - runtime constants
public class Configuration
{
    public readonly string ConnectionString;
    public readonly DateTime StartTime = DateTime.Now;
    
    public Configuration(string connStr)
    {
        ConnectionString = connStr; // Can only be set in constructor
    }
}

Operators 

// Arithmetic Operators
int a = 10, b = 3;
int sum = a + b;        // 13
int difference = a - b; // 7
int product = a * b;    // 30
int quotient = a / b;   // 3 (integer division)
int remainder = a % b;  // 1

// Increment/Decrement
int x = 5;
int preInc = ++x;  // x = 6, preInc = 6
int postInc = x++; // postInc = 6, x = 7

// Comparison Operators
bool isEqual = (a == b);     // false
bool isNotEqual = (a != b);  // true
bool isGreater = (a > b);    // true

// Logical Operators
bool result1 = true && false;  // false (AND)
bool result2 = true || false;  // true (OR)
bool result3 = !true;          // false (NOT)

// Null-coalescing Operators
string name = null;
string displayName = name ?? "Unknown";           // "Unknown"
string shortName = name?.Substring(0, 3) ?? "N/A"; // "N/A"

Control Structures

Conditional Statements
// If-else statement
int score = 85;
string grade;

if (score >= 90)
    grade = "A";
else if (score >= 80)
    grade = "B";
else if (score >= 70)
    grade = "C";
else
    grade = "F";

// Switch expression (C# 8.0+)
string GetGrade(int score) => score switch
{
    >= 90 => "A",
    >= 80 => "B",
    >= 70 => "C",
    _ => "F"
};

// Ternary operator
string status = age >= 18 ? "Adult" : "Minor";
Loops
// For loop
for (int i = 0; i < 5; i++)
{
    Console.WriteLine($"Count: {i}");
}

// While loop
int count = 0;
while (count < 3)
{
    Console.WriteLine($"While: {count}");
    count++;
}

// Do-while loop
do
{
    Console.WriteLine("Executed at least once");
    count--;
} while (count > 0);

// Foreach loop
string[] names = {"Alice", "Bob", "Charlie"};
foreach (string name in names)
{
    Console.WriteLine($"Hello, {name}!");
}

Methods

public class Calculator
{
    // Basic method
    public int Add(int a, int b)
    {
        return a + b;
    }
    
    // Method with optional parameters
    public void PrintMessage(string message, bool uppercase = false)
    {
        if (uppercase)
            Console.WriteLine(message.ToUpper());
        else
            Console.WriteLine(message);
    }
    
    // Method with params keyword
    public int Sum(params int[] numbers)
    {
        int total = 0;
        foreach (int num in numbers)
        {
            total += num;
        }
        return total;
    }
    
    // Method overloading
    public double Add(double a, double b) => a + b;
    public string Add(string a, string b) => a + b;
    
    // Ref and out parameters
    public void SwapNumbers(ref int a, ref int b)
    {
        int temp = a;
        a = b;
        b = temp;
    }
    
    public bool TryParseAge(string input, out int age)
    {
        return int.TryParse(input, out age) && age >= 0;
    }
}

Classes & Objects 

public class Student
{
    // Fields
    private string name;
    private int age;
    private List<string> courses;
    
    // Properties
    public string Name 
    { 
        get { return name; } 
        set { name = value?.Trim(); } 
    }
    
    public int Age 
    { 
        get { return age; } 
        set 
        { 
            if (value >= 0 && value <= 150)
                age = value;
            else
                throw new ArgumentException("Invalid age");
        } 
    }
    
    // Auto-implemented property
    public string StudentId { get; set; }
    
    // Constructors
    public Student() // Default constructor
    {
        courses = new List<string>();
    }
    
    public Student(string name, int age) : this() // Constructor chaining
    {
        Name = name;
        Age = age;
        StudentId = Guid.NewGuid().ToString();
    }
    
    // Methods
    public void EnrollInCourse(string course)
    {
        if (!courses.Contains(course))
        {
            courses.Add(course);
            Console.WriteLine($"{Name} enrolled in {course}");
        }
    }
    
    public void DisplayInfo()
    {
        Console.WriteLine($"Student: {Name}, Age: {Age}, ID: {StudentId}");
        Console.WriteLine($"Courses: {string.Join(", ", courses)}");
    }
    
    // Static method
    public static Student CreateFromString(string data)
    {
        var parts = data.Split(',');
        return new Student(parts[0], int.Parse(parts[1]));
    }
}

Practice Questions

Answer:

  • Value Types: Store actual data in memory. Examples: int, double, bool, struct, enum. When assigned, the actual value is copied.
  • Reference Types: Store references (addresses) to data. Examples: string, object, class, array. When assigned, only the reference is copied, not the actual data. 
// Value type example
int a = 10;
int b = a; // b gets a copy of a's value
a = 20;    // b is still 10

// Reference type example
int[] arr1 = {1, 2, 3};
int[] arr2 = arr1; // arr2 points to same array as arr1
arr1[0] = 100;     // arr2[0] is also 100 now

Answer: Method overloading allows multiple methods with the same name but different parameters (type, number, or order).

public class FactorialCalculator
{
    // Iterative approach
    public static long FactorialIterative(int n)
    {
        if (n < 0) throw new ArgumentException("Negative numbers not allowed");
        if (n == 0 || n == 1) return 1;
        
        long result = 1;
        for (int i = 2; i <= n; i++)
        {
            result *= i;
        }
        return result;
    }
    
    // Recursive approach
    public static long FactorialRecursive(int n)
    {
        if (n < 0) throw new ArgumentException("Negative numbers not allowed");
        if (n == 0 || n == 1) return 1;
        
        return n * FactorialRecursive(n - 1);
    }
    
    static void Main()
    {
        Console.Write("Enter a number: ");
        if (int.TryParse(Console.ReadLine(), out int number))
        {
            try
            {
                long result = FactorialIterative(number);
                Console.WriteLine($"Factorial of {number} is {result}");
            }
            catch (ArgumentException ex)
            {
                Console.WriteLine(ex.Message);
            }
        }
        else
        {
            Console.WriteLine("Invalid input. Please enter a valid number.");
        }
    }
}
