Operators in C#

Learn about different types of operators and their usage in C#

Understanding Operators

Symbols that perform specific operations on operands

Operators are symbols that perform specific operations on one or more operands. C# provides a rich set of operators for arithmetic, comparison, logical, assignment, and other operations.

Types of Operators:
  • Arithmetic Operators: +, -, *, /, % (modulo)
  • Comparison Operators: ==, !=, <, >, <=, >=
  • Logical Operators: && (AND), || (OR), ! (NOT)
  • Assignment Operators: =, +=, -=, *=, /=, %=
  • Increment/Decrement: ++, --
  • Bitwise Operators: &, |, ^, ~, <<, >>

Arithmetic Operators

Basic mathematical operations

using System;

class ArithmeticOperators
{
    static void Main()
    {
        Console.WriteLine("=== Arithmetic Operators ===");
        
        int a = 15, b = 4;
        Console.WriteLine($"a = {a}, b = {b}");
        
        // Basic arithmetic operations
        Console.WriteLine($"Addition (a + b): {a + b}");
        Console.WriteLine($"Subtraction (a - b): {a - b}");
        Console.WriteLine($"Multiplication (a * b): {a * b}");
        Console.WriteLine($"Division (a / b): {a / b}");
        Console.WriteLine($"Modulus (a % b): {a % b}");
        
        // Floating-point division
        double x = 15.0, y = 4.0;
        Console.WriteLine($"Floating-point division ({x} / {y}): {x / y}");
        
        // Increment and Decrement
        int count = 10;
        Console.WriteLine($"Original count: {count}");
        Console.WriteLine($"Pre-increment (++count): {++count}");
        Console.WriteLine($"Post-increment (count++): {count++}");
        Console.WriteLine($"After post-increment: {count}");
        
        Console.WriteLine($"Pre-decrement (--count): {--count}");
        Console.WriteLine($"Post-decrement (count--): {count--}");
        Console.WriteLine($"After post-decrement: {count}");
    }
}

Comparison Operators

Comparing values and returning boolean results

using System;

class ComparisonOperators
{
    static void Main()
    {
        Console.WriteLine("=== Comparison Operators ===");
        
        int x = 10, y = 20;
        Console.WriteLine($"x = {x}, y = {y}");
        
        Console.WriteLine($"Equal (x == y): {x == y}");
        Console.WriteLine($"Not Equal (x != y): {x != y}");
        Console.WriteLine($"Greater than (x > y): {x > y}");
        Console.WriteLine($"Less than (x < y): {x < y}");
        Console.WriteLine($"Greater than or equal (x >= y): {x >= y}");
        Console.WriteLine($"Less than or equal (x <= y): {x <= y}");
        
        // String comparison
        string str1 = "Hello";
        string str2 = "World";
        string str3 = "Hello";
        
        Console.WriteLine($"\nString comparison:");
        Console.WriteLine($"str1 == str2: {str1 == str2}");
        Console.WriteLine($"str1 == str3: {str1 == str3}");
        
        // Reference vs Value comparison
        object obj1 = str1;
        object obj2 = str3;
        Console.WriteLine($"ReferenceEquals(obj1, obj2): {ReferenceEquals(obj1, obj2)}");
    }
}

Logical Operators

Boolean logic operations

using System;

class LogicalOperators
{
    static void Main()
    {
        Console.WriteLine("=== Logical Operators ===");
        
        bool p = true, q = false;
        Console.WriteLine($"p = {p}, q = {q}");
        
        // Basic logical operations
        Console.WriteLine($"Logical AND (p && q): {p && q}");
        Console.WriteLine($"Logical OR (p || q): {p || q}");
        Console.WriteLine($"Logical NOT (!p): {!p}");
        Console.WriteLine($"Logical NOT (!q): {!q}");
        
        // Short-circuit evaluation
        Console.WriteLine("\n=== Short-circuit Evaluation ===");
        int a = 5, b = 0;
        
        // Short-circuit AND - second condition not evaluated if first is false
        if (b != 0 && a / b > 2)
        {
            Console.WriteLine("This won't execute because b == 0");
        }
        else
        {
            Console.WriteLine("Safe: First condition (b != 0) prevented division by zero");
        }
        
        // Short-circuit OR - second condition not evaluated if first is true
        if (a > 0 || b / a > 1) // b/a won't be evaluated because a > 0 is true
        {
            Console.WriteLine("First condition was true, second not evaluated");
        }
        
        // Complex logical expressions
        int age = 25;
        bool hasLicense = true;
        bool hasInsurance = true;
        
        if (age >= 18 && hasLicense && hasInsurance)
        {
            Console.WriteLine("Can drive legally");
        }
        
        if (age < 16 || !hasLicense || !hasInsurance)
        {
            Console.WriteLine("Cannot drive or needs requirements");
        }
    }
}

Assignment Operators

Assigning and modifying values

using System;

class AssignmentOperators
{
    static void Main()
    {
        Console.WriteLine("=== Assignment Operators ===");
        
        int val = 10;
        Console.WriteLine($"Initial value: {val}");
        
        // Compound assignment operators
        val += 5; // val = val + 5
        Console.WriteLine($"After += 5: {val}");
        
        val -= 3; // val = val - 3
        Console.WriteLine($"After -= 3: {val}");
        
        val *= 2; // val = val * 2
        Console.WriteLine($"After *= 2: {val}");
        
        val /= 4; // val = val / 4
        Console.WriteLine($"After /= 4: {val}");
        
        val %= 3; // val = val % 3
        Console.WriteLine($"After %= 3: {val}");
        
        // Multiple assignments
        int a, b, c;
        a = b = c = 5; // Right to left assignment
        Console.WriteLine($"Multiple assignment: a={a}, b={b}, c={c}");
        
        // Null-coalescing assignment (C# 8.0+)
        string name = null;
        name ??= "Default Name"; // Assign only if null
        Console.WriteLine($"After ??= assignment: {name}");
        
        name ??= "Another Name"; // Won't assign because name is not null
        Console.WriteLine($"After second ??=: {name}");
    }
}

Bitwise Operators

Working with individual bits

using System;

class BitwiseOperators
{
    static void Main()
    {
        Console.WriteLine("=== Bitwise Operators ===");
        
        int m = 12; // Binary: 1100
        int n = 10; // Binary: 1010
        
        Console.WriteLine($"m = {m} (Binary: {Convert.ToString(m, 2).PadLeft(8, '0')})");
        Console.WriteLine($"n = {n} (Binary: {Convert.ToString(n, 2).PadLeft(8, '0')})");
        
        // Bitwise operations
        int andResult = m & n;
        int orResult = m | n;
        int xorResult = m ^ n;
        int notResult = ~m;
        int leftShift = m << 1;
        int rightShift = m >> 1;
        
        Console.WriteLine($"Bitwise AND (m & n): {andResult} (Binary: {Convert.ToString(andResult, 2).PadLeft(8, '0')})");
        Console.WriteLine($"Bitwise OR (m | n): {orResult} (Binary: {Convert.ToString(orResult, 2).PadLeft(8, '0')})");
        Console.WriteLine($"Bitwise XOR (m ^ n): {xorResult} (Binary: {Convert.ToString(xorResult, 2).PadLeft(8, '0')})");
        Console.WriteLine($"Bitwise NOT (~m): {notResult}");
        Console.WriteLine($"Left Shift (m << 1): {leftShift} (Binary: {Convert.ToString(leftShift, 2).PadLeft(8, '0')})");
        Console.WriteLine($"Right Shift (m >> 1): {rightShift} (Binary: {Convert.ToString(rightShift, 2).PadLeft(8, '0')})");
        
        // Practical example: Flags enumeration
        Console.WriteLine("\n=== Practical Example: File Permissions ===");
        
        const int READ = 1;    // 001
        const int WRITE = 2;   // 010
        const int EXECUTE = 4; // 100
        
        int permissions = READ | WRITE; // Combine permissions
        Console.WriteLine($"Initial permissions: {permissions}");
        
        // Check if has read permission
        bool hasRead = (permissions & READ) != 0;
        Console.WriteLine($"Has read permission: {hasRead}");
        
        // Add execute permission
        permissions |= EXECUTE;
        Console.WriteLine($"After adding execute: {permissions}");
        
        // Remove write permission
        permissions &= ~WRITE;
        Console.WriteLine($"After removing write: {permissions}");
    }
}

Special Operators

Conditional, null-coalescing, and other special operators

using System;

class SpecialOperators
{
    static void Main()
    {
        Console.WriteLine("=== Special Operators ===");
        
        // Conditional (Ternary) Operator
        int age = 18;
        string status = age >= 18 ? "Adult" : "Minor";
        Console.WriteLine($"Age: {age}, Status: {status}");
        
        // Null-coalescing Operators
        string nullString = null;
        string defaultString = "Default Value";
        
        string result1 = nullString ?? defaultString;
        Console.WriteLine($"nullString ?? defaultString: {result1}");
        
        string result2 = "Not Null" ?? defaultString;
        Console.WriteLine($"\"Not Null\" ?? defaultString: {result2}");
        
        // Null-conditional Operators
        string name = "John Doe";
        int? length = name?.Length; // Safe navigation
        Console.WriteLine($"Name length: {length}");
        
        name = null;
        length = name?.Length; // Returns null instead of throwing exception
        Console.WriteLine($"Null name length: {length}");
        
        // Type checking operators
        object obj = "Hello World";
        
        // is operator
        if (obj is string)
        {
            Console.WriteLine("obj is a string");
        }
        
        // is with pattern matching (C# 7.0+)
        if (obj is string str)
        {
            Console.WriteLine($"String value: {str}");
        }
        
        // as operator
        string objAsString = obj as string;
        if (objAsString != null)
        {
            Console.WriteLine($"Converted to string: {objAsString}");
        }
        
        // typeof operator
        Type stringType = typeof(string);
        Console.WriteLine($"Type of string: {stringType.Name}");
        
        // sizeof operator (for value types)
        Console.WriteLine($"Size of int: {sizeof(int)} bytes");
        Console.WriteLine($"Size of double: {sizeof(double)} bytes");
    }
}
