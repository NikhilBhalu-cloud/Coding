Classes & Objects

Understanding the fundamental concepts of object-oriented programming

What are Classes and Objects?

Building blocks of object-oriented programming

Class

A class is a blueprint or template that defines the structure and behavior of objects.

  • Defines data members (fields)
  • Defines behavior (methods)
  • Encapsulates data and methods
  • Can be instantiated multiple times
Object

An object is an instance of a class that contains actual data.

  • Created from a class template
  • Has its own memory space
  • Contains actual values
  • Can interact with other objects

Basic Class Definition

Creating your first class with fields, properties, and methods

using System;

// Basic class definition
public class Person
{
    // Private fields (data encapsulation)
    private string _name;
    private int _age;
    
    // Properties (controlled access to data)
    public string Name
    {
        get { return _name; }
        set 
        {
            if (string.IsNullOrWhiteSpace(value))
                throw new ArgumentException("Name cannot be empty");
            _name = value;
        }
    }
    
    public int Age
    {
        get { return _age; }
        set 
        {
            if (value < 0 || value > 150)
                throw new ArgumentException("Age must be between 0 and 150");
            _age = value;
        }
    }
    
    // Constructor - initializes new objects
    public Person(string name, int age)
    {
        Name = name;  // Using property for validation
        Age = age;    // Using property for validation
    }
    
    // Method - defines behavior
    public void DisplayInfo()
    {
        Console.WriteLine($"Name: {Name}, Age: {Age}");
    }
    
    // Method with return value
    public string GetGreeting()
    {
        return $"Hello, my name is {Name} and I'm {Age} years old.";
    }
}

// Using the class
class Program
{
    static void Main()
    {
        // Creating objects (instances) of the Person class
        Person person1 = new Person("Alice", 25);
        Person person2 = new Person("Bob", 30);
        
        // Using object methods
        person1.DisplayInfo();
        person2.DisplayInfo();
        
        // Getting values from objects
        Console.WriteLine(person1.GetGreeting());
        Console.WriteLine(person2.GetGreeting());
    }
}

Constructors

Special methods for object initialization

public class Car
{
    public string Make { get; set; }
    public string Model { get; set; }
    public int Year { get; set; }
    public string Color { get; set; }
    
    // Default constructor
    public Car()
    {
        Make = "Unknown";
        Model = "Unknown";
        Year = DateTime.Now.Year;
        Color = "White";
    }
    
    // Parameterized constructor
    public Car(string make, string model)
    {
        Make = make;
        Model = model;
        Year = DateTime.Now.Year;
        Color = "White";
    }
    
    // Constructor with all parameters
    public Car(string make, string model, int year, string color)
    {
        Make = make;
        Model = model;
        Year = year;
        Color = color;
    }
    
    // Copy constructor
    public Car(Car other)
    {
        Make = other.Make;
        Model = other.Model;
        Year = other.Year;
        Color = other.Color;
    }
    
    public void DisplayInfo()
    {
        Console.WriteLine($"{Year} {Color} {Make} {Model}");
    }
}

class ConstructorExample
{
    static void Main()
    {
        // Using different constructors
        Car car1 = new Car();                              // Default constructor
        Car car2 = new Car("Toyota", "Camry");             // Parameterized constructor
        Car car3 = new Car("Honda", "Civic", 2023, "Red"); // Full constructor
        Car car4 = new Car(car3);                          // Copy constructor
        
        car1.DisplayInfo();
        car2.DisplayInfo(); 
        car3.DisplayInfo();
        car4.DisplayInfo();
    }
}

Properties vs Fields

Understanding data encapsulation and controlled access

public class BankAccount
{
    // Private fields - not accessible from outside
    private string _accountNumber;
    private decimal _balance;
    private DateTime _lastTransactionDate;
    
    // Public properties - controlled access
    public string AccountNumber
    {
        get { return _accountNumber; }
        private set { _accountNumber = value; } // Private setter
    }
    
    public decimal Balance
    {
        get { return _balance; }
        set 
        {
            if (value < 0)
                throw new ArgumentException("Balance cannot be negative");
            _balance = value;
            _lastTransactionDate = DateTime.Now;
        }
    }
    
    // Read-only property
    public DateTime LastTransactionDate
    {
        get { return _lastTransactionDate; }
    }
    
    // Auto-implemented property
    public string AccountHolder { get; set; }
    
    // Calculated property
    public string AccountSummary => $"Account: {AccountNumber}, Balance: ${Balance:F2}";
    
    public BankAccount(string accountNumber, string accountHolder, decimal initialBalance)
    {
        AccountNumber = accountNumber;
        AccountHolder = accountHolder;
        Balance = initialBalance;
    }
    
    public void Deposit(decimal amount)
    {
        if (amount <= 0)
            throw new ArgumentException("Deposit amount must be positive");
        
        Balance += amount; // Uses property setter
        Console.WriteLine($"Deposited ${amount:F2}. New balance: ${Balance:F2}");
    }
    
    public bool Withdraw(decimal amount)
    {
        if (amount <= 0)
            throw new ArgumentException("Withdrawal amount must be positive");
        
        if (amount > Balance)
        {
            Console.WriteLine("Insufficient funds");
            return false;
        }
        
        Balance -= amount; // Uses property setter
        Console.WriteLine($"Withdrew ${amount:F2}. New balance: ${Balance:F2}");
        return true;
    }
}

class PropertyExample
{
    static void Main()
    {
        BankAccount account = new BankAccount("12345", "John Doe", 1000);
        
        Console.WriteLine(account.AccountSummary);
        
        account.Deposit(500);
        account.Withdraw(200);
        
        Console.WriteLine($"Last transaction: {account.LastTransactionDate}");
    }
}

Static Members

Class-level data and methods shared by all instances

public class Student
{
    // Static field - shared by all instances
    private static int _nextId = 1;
    public static int TotalStudents { get; private set; } = 0;
    
    // Instance fields
    public int Id { get; private set; }
    public string Name { get; set; }
    public string Grade { get; set; }
    
    // Static property
    public static string SchoolName { get; set; } = "C# Learning Academy";
    
    public Student(string name, string grade)
    {
        Id = _nextId++;
        Name = name;
        Grade = grade;
        TotalStudents++; // Increment static counter
    }
    
    // Instance method
    public void DisplayInfo()
    {
        Console.WriteLine($"Student ID: {Id}, Name: {Name}, Grade: {Grade}");
    }
    
    // Static method
    public static void DisplaySchoolInfo()
    {
        Console.WriteLine($"School: {SchoolName}");
        Console.WriteLine($"Total Students: {TotalStudents}");
    }
    
    // Static method with parameters
    public static Student CreateHonorStudent(string name)
    {
        return new Student(name, "A+");
    }
}

class StaticExample
{
    static void Main()
    {
        // Access static members without creating instances
        Console.WriteLine($"School: {Student.SchoolName}");
        Console.WriteLine($"Total students initially: {Student.TotalStudents}");
        
        // Create instances
        Student student1 = new Student("Alice", "A");
        Student student2 = new Student("Bob", "B");
        Student student3 = Student.CreateHonorStudent("Charlie");
        
        // Display instance information
        student1.DisplayInfo();
        student2.DisplayInfo();
        student3.DisplayInfo();
        
        // Display static information
        Student.DisplaySchoolInfo();
    }
}
