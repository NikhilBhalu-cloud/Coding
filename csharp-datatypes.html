Data Types & Variables

Understanding C# data types and variable declarations

Understanding Data Types

C# is a strongly-typed language with a rich type system

C# is a strongly-typed language where every variable must have a declared type. Data types specify the type of data that variables can store.

Value Types

Store data directly in memory. Examples:

  • int - 32-bit integer
  • double - 64-bit floating point
  • bool - Boolean (true/false)
  • char - Single Unicode character
  • decimal - 128-bit decimal
Reference Types

Store references to data. Examples:

  • string - Sequence of characters
  • object - Base type for all types
  • arrays - Collection of elements
  • class - Custom reference types

Value Types

Exploring integer, floating-point, and other value types

using System;

class ValueTypesExample
{
    static void Main()
    {
        Console.WriteLine("=== Value Types ===");
        
        // Integer types
        byte byteVar = 255;
        short shortVar = -32768;
        int intVar = 2147483647;
        long longVar = 9223372036854775807L;
        
        Console.WriteLine($"byte: {byteVar} (Range: 0 to 255)");
        Console.WriteLine($"short: {shortVar} (Range: -32,768 to 32,767)");
        Console.WriteLine($"int: {intVar} (Range: -2,147,483,648 to 2,147,483,647)");
        Console.WriteLine($"long: {longVar} (Range: very large)");
        
        // Floating-point types
        float floatVar = 3.14159f;
        double doubleVar = 3.141592653589793;
        decimal decimalVar = 123.456789m;
        
        Console.WriteLine($"\nfloat: {floatVar} (7 digits precision)");
        Console.WriteLine($"double: {doubleVar} (15-17 digits precision)");
        Console.WriteLine($"decimal: {decimalVar} (28-29 digits precision)");
        
        // Other value types
        bool boolVar = true;
        char charVar = 'A';
        
        Console.WriteLine($"\nbool: {boolVar}");
        Console.WriteLine($"char: {charVar}");
    }
}

Reference Types

Working with strings, objects, and arrays

using System;

class ReferenceTypesExample
{
    static void Main()
    {
        Console.WriteLine("=== Reference Types ===");
        
        // String type
        string name = "John Doe";
        string message = "Hello, C#!";
        
        Console.WriteLine($"string name: {name}");
        Console.WriteLine($"string message: {message}");
        
        // Object type
        object obj1 = "This is a string stored in object";
        object obj2 = 42;
        object obj3 = true;
        
        Console.WriteLine($"object containing string: {obj1}");
        Console.WriteLine($"object containing int: {obj2}");
        Console.WriteLine($"object containing bool: {obj3}");
        
        // Array type
        int[] numbers = {1, 2, 3, 4, 5};
        string[] colors = {"Red", "Green", "Blue"};
        
        Console.WriteLine($"int array: [{string.Join(", ", numbers)}]");
        Console.WriteLine($"string array: [{string.Join(", ", colors)}]");
    }
}

Type Inference with var

Let the compiler determine the type automatically

using System;

class TypeInferenceExample
{
    static void Main()
    {
        Console.WriteLine("=== Type Inference (var) ===");
        
        // var keyword - compiler infers the type
        var implicitInt = 42;
        var implicitString = "Inferred as string";
        var implicitDouble = 3.14;
        var implicitBool = true;
        
        Console.WriteLine($"var implicitInt = {implicitInt} (Type: {implicitInt.GetType().Name})");
        Console.WriteLine($"var implicitString = \"{implicitString}\" (Type: {implicitString.GetType().Name})");
        Console.WriteLine($"var implicitDouble = {implicitDouble} (Type: {implicitDouble.GetType().Name})");
        Console.WriteLine($"var implicitBool = {implicitBool} (Type: {implicitBool.GetType().Name})");
        
        // Complex types with var
        var numbers = new int[] {1, 2, 3};
        var person = new { Name = "Alice", Age = 30 }; // Anonymous type
        
        Console.WriteLine($"var numbers type: {numbers.GetType().Name}");
        Console.WriteLine($"Anonymous type: {person.Name}, {person.Age}");
    }
}

Nullable Types

Allowing value types to represent null values

using System;

class NullableTypesExample
{
    static void Main()
    {
        Console.WriteLine("=== Nullable Types ===");
        
        // Nullable value types
        int? nullableInt = null;
        double? nullableDouble = 3.14;
        bool? nullableBool = true;
        
        Console.WriteLine($"int? nullableInt = {nullableInt?.ToString() ?? "null"}");
        Console.WriteLine($"double? nullableDouble = {nullableDouble}");
        Console.WriteLine($"bool? nullableBool = {nullableBool}");
        
        // Checking for null
        if (nullableInt.HasValue)
        {
            Console.WriteLine($"nullableInt has value: {nullableInt.Value}");
        }
        else
        {
            Console.WriteLine("nullableInt is null");
        }
        
        // Null-coalescing operator
        int actualValue = nullableInt ?? 0; // Use 0 if null
        Console.WriteLine($"Using null-coalescing: {actualValue}");
        
        // Assign value
        nullableInt = 42;
        Console.WriteLine($"After assignment: {nullableInt}");
    }
}

Type Conversion

Converting between different data types

using System;

class TypeConversionExample
{
    static void Main()
    {
        Console.WriteLine("=== Type Conversion ===");
        
        // Implicit conversion (safe, no data loss)
        int intValue = 100;
        long longValue = intValue; // int to long
        double doubleValue = intValue; // int to double
        
        Console.WriteLine($"int: {intValue}");
        Console.WriteLine($"long: {longValue}");
        Console.WriteLine($"double: {doubleValue}");
        
        // Explicit conversion (casting, potential data loss)
        double sourceDouble = 123.456;
        int targetInt = (int)sourceDouble; // Truncates decimal part
        
        Console.WriteLine($"Original double: {sourceDouble}");
        Console.WriteLine($"Converted to int: {targetInt}");
        
        // Using Convert class
        string numberString = "789";
        int convertedInt = Convert.ToInt32(numberString);
        double convertedDouble = Convert.ToDouble(numberString);
        
        Console.WriteLine($"String: \"{numberString}\"");
        Console.WriteLine($"Converted to int: {convertedInt}");
        Console.WriteLine($"Converted to double: {convertedDouble}");
        
        // Parse methods
        string boolString = "true";
        bool parsedBool = bool.Parse(boolString);
        
        Console.WriteLine($"Parsed bool: {parsedBool}");
    }
}
